/* Q. Create a Queue using Linked List in Java and perform the following operations:
Insert elements 10, 20, 30, 40 into the queue.
Display the queue.
Remove one element from the queue.
Display the queue again.
Show the front element using peek().
Check whether the queue is empty.
 */


class Node {
    int data;
    Node next;

    Node(int data) {
        this.data = data;
        this.next = null;
    }
}

class QueueLL {

    Node front;
    Node rear;


    void enqueue(int data) {

        Node newNode = new Node(data);

        if (rear == null) {
            front = rear = newNode;
            return;
        }

        rear.next = newNode;
        rear = newNode;
    }


    void dequeue() {

        if (front == null) {
            System.out.println("Queue is empty");
            return;
        }

        System.out.println("Removed element: " + front.data);
        front = front.next;

        if (front == null)
            rear = null;
    }


    int peek() {

        if (front == null) {
            System.out.println("Queue is empty");
            return -1;
        }

        return front.data;
    }


    boolean isEmpty() {
        return front == null;
    }


    void display() {

        if (front == null) {
            System.out.println("Queue is empty");
            return;
        }

        Node temp = front;

        System.out.print("Queue: ");

        while (temp != null) {
            System.out.print(temp.data + " ");
            temp = temp.next;
        }

        System.out.println();
    }
}

 class MainM {

    public static void main(String[] args) {

        QueueLL q = new QueueLL();

        q.enqueue(10);
        q.enqueue(20);
        q.enqueue(30);
        q.enqueue(40);

        q.display();

        q.dequeue();

        q.display();

        System.out.println("Front element: " + q.peek());

        System.out.println("Is queue empty? " + q.isEmpty());
    }
}
